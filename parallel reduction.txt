#include <iostream>
#include <vector>
#include <climits>
#include <omp.h>

using namespace std;

int main() {

    vector<int> arr = {10, 20, 5, 30, 15};

    int min_val = INT_MAX;
    int max_val = INT_MIN;
    int sum = 0;

    // Parallel reduction
    #pragma omp parallel for reduction(min:min_val) reduction(max:max_val) reduction(+:sum)
    for(int i = 0; i < arr.size(); i++) {

        if(arr[i] < min_val)
            min_val = arr[i];

        if(arr[i] > max_val)
            max_val = arr[i];

        sum += arr[i];
    }

    double avg = (double)sum / arr.size();

    cout << "Minimum = " << min_val << endl;
    cout << "Maximum = " << max_val << endl;
    cout << "Sum = " << sum << endl;
    cout << "Average = " << avg << endl;

    return 0;
}
